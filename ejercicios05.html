c.Â Implemente y grafique la funciÃ³n log(L(\(\theta\))).

#Se realiza la funciÃ³n de verosimilitud
log_likelihood <- function(theta, Y, X){
  X <- as.matrix(X); Y <- as.matrix(Y);
  N <- nrow(X)
  beta <- theta[1]
  sigma_2 <- theta[2]
  e <- Y - beta*X
  loglik <- -0.5*N*log(2*pi) - 0.5*N*log(sigma_2) - ((t(e) %*% e)/(2*sigma_2))
  return(-loglik)
}

#Se grafica la funciÃ³n de verosimilitud
log_like_graph <- function(beta, sigma_2){

}
